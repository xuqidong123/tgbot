import sqlite3
import os

DB_PATH = os.path.join(os.path.dirname(os.path.abspath(__file__)), "bot.db")

def get_conn():
    return sqlite3.connect(DB_PATH)

def init_db():
    conn = get_conn()
    c = conn.cursor()
    c.execute('''CREATE TABLE IF NOT EXISTS users (
        user_id INTEGER PRIMARY KEY,
        username TEXT,
        api_key TEXT,
        api_secret TEXT,
        leverage INTEGER DEFAULT 10,
        amount REAL DEFAULT 100,
        active INTEGER DEFAULT 1,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS trades (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        user_id INTEGER,
        symbol TEXT,
        action TEXT,
        price REAL,
        amount REAL,
        leverage INTEGER,
        status TEXT,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS signal_coins (
        coin TEXT PRIMARY KEY,
        last_signal TEXT,
        last_time TEXT,
        signal_count INTEGER DEFAULT 0
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS user_coins (
        user_id INTEGER,
        coin TEXT,
        enabled INTEGER DEFAULT 1,
        amount REAL DEFAULT 0,
        PRIMARY KEY(user_id, coin)
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS coin_intervals (
        coin TEXT,
        interval TEXT,
        first_seen TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        PRIMARY KEY(coin, interval)
    )''')
    c.execute('''CREATE TABLE IF NOT EXISTS alert_groups (
        group_id TEXT PRIMARY KEY,
        group_name TEXT DEFAULT '',
        alert_open INTEGER DEFAULT 1,
        alert_fail INTEGER DEFAULT 1,
        alert_error INTEGER DEFAULT 1,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )''')

    for stmt in [
        "ALTER TABLE users ADD COLUMN mode TEXT DEFAULT 'both'",
        "ALTER TABLE users ADD COLUMN tp_type TEXT DEFAULT 'none'",
        "ALTER TABLE users ADD COLUMN tp_value REAL DEFAULT 0",
        "ALTER TABLE users ADD COLUMN sl_type TEXT DEFAULT 'none'",
        "ALTER TABLE users ADD COLUMN sl_value REAL DEFAULT 0",
        "ALTER TABLE users ADD COLUMN exchange TEXT DEFAULT 'binance'",
        "ALTER TABLE users ADD COLUMN lighter_private_key TEXT",
        "ALTER TABLE users ADD COLUMN lighter_account_index INTEGER DEFAULT 0",
        "ALTER TABLE users ADD COLUMN lighter_api_key_index INTEGER DEFAULT 0",
        "ALTER TABLE user_coins ADD COLUMN interval TEXT DEFAULT ''",
    ]:
        try:
            c.execute(stmt)
        except:
            pass

    conn.commit()
    conn.close()

def _row_to_user(row):
    if not row:
        return None
    user = {
        "user_id": row[0], "username": row[1],
        "api_key": row[2], "api_secret": row[3],
        "leverage": row[4], "amount": row[5], "active": row[6],
    }
    user["mode"] = row[8] if len(row) > 8 else "both"
    user["tp_type"] = row[9] if len(row) > 9 else "none"
    user["tp_value"] = row[10] if len(row) > 10 else 0
    user["sl_type"] = row[11] if len(row) > 11 else "none"
    user["sl_value"] = row[12] if len(row) > 12 else 0
    user["exchange"] = row[13] if len(row) > 13 else "binance"
    user["lighter_private_key"] = row[14] if len(row) > 14 else None
    user["lighter_account_index"] = row[15] if len(row) > 15 else 0
    user["lighter_api_key_index"] = row[16] if len(row) > 16 else 0
    return user

def get_exchange(api_key, api_secret):
    exchange = ccxt.binanceusdm({
        "apiKey": api_key,
        "secret": api_secret,
        "enableRateLimit": True,
        "proxies": {
            "http": PROXY,
            "https": PROXY,
        },
        "aiohttp_proxy": PROXY,
    })
    if BINANCE_TESTNET:
        exchange.enable_demo_trading(True)

def get_balance(api_key, api_secret):
    try:
        exchange = get_exchange(api_key, api_secret)
        balance = exchange.fetch_balance()
        usdt = balance.get("USDT", {})
        return {
            "total": usdt.get("total", 0),
            "free": usdt.get("free", 0),
            "used": usdt.get("used", 0),
        }
    except Exception as e:
        return {"error": str(e)}

def get_account_info(exchange):
    try:
        balance = exchange.fetch_balance()
        usdt = balance.get("USDT", {})
        total = float(usdt.get("total", 0))
        free = float(usdt.get("free", 0))
        used = float(usdt.get("used", 0))
        return {"total": round(total, 2), "free": round(free, 2), "used": round(used, 2)}
    except:
        return {"total": 0, "free": 0, "used": 0}

def get_current_position(exchange, symbol):
    try:
        positions = exchange.fetch_positions([symbol])
        for pos in positions:
            amt = float(pos["contracts"])
            if amt > 0:
                return {
                    "side": pos["side"],
                    "amount": amt,
                    "entry_price": pos["entryPrice"],
                    "pnl": float(pos["unrealizedPnl"] or 0),
                    "notional": float(pos.get("notional") or 0),
                    "margin": float(pos.get("initialMargin") or 0),
                }
        return None
    except:
        return None

def get_trade_fees(exchange, symbol, limit=5):
    try:
        trades = exchange.fetch_my_trades(symbol, limit=limit)
        total_fee = 0
        for t in trades:
            fee = t.get("fee", {})
            total_fee += float(fee.get("cost", 0))
        return round(total_fee, 6)
    except:
        return 0

def calc_tp_sl_price(action, entry_price, tp_type, tp_value, sl_type, sl_value):
    tp_price = None
    sl_price = None
    if tp_type == "percent" and tp_value > 0:
        if action == "buy":
            tp_price = entry_price * (1 + tp_value / 100)
        else:
            tp_price = entry_price * (1 - tp_value / 100)
    elif tp_type == "amount" and tp_value > 0:
        if action == "buy":
            tp_price = entry_price + tp_value
        else:
            tp_price = entry_price - tp_value

    if sl_type == "percent" and sl_value > 0:
        if action == "buy":
            sl_price = entry_price * (1 - sl_value / 100)
        else:
            sl_price = entry_price * (1 + sl_value / 100)
    elif sl_type == "amount" and sl_value > 0:
        if action == "buy":
            sl_price = entry_price - sl_value
        else:
            sl_price = entry_price + sl_value

    return tp_price, sl_price

def place_tp_sl_orders(exchange, symbol, action, amount, tp_price, sl_price):
    tp_sl_info = []
    try:
        if tp_price:
            tp_price = float(exchange.price_to_precision(symbol, tp_price))
            close_side = "sell" if action == "buy" else "buy"
            exchange.create_order(symbol, "TAKE_PROFIT_MARKET", close_side, amount, None, {
                "stopPrice": tp_price,
                "closePosition": False,
                "reduceOnly": True,
            })
            tp_sl_info.append(f"止盈价: {tp_price}")
    except Exception as e:
        tp_sl_info.append(f"止盈设置失败: {e}")

    try:
        if sl_price:
            sl_price = float(exchange.price_to_precision(symbol, sl_price))
            close_side = "sell" if action == "buy" else "buy"
            exchange.create_order(symbol, "STOP_MARKET", close_side, amount, None, {
                "stopPrice": sl_price,
                "closePosition": False,
                "reduceOnly": True,
            })
            tp_sl_info.append(f"止损价: {sl_price}")
    except Exception as e:
        tp_sl_info.append(f"止损设置失败: {e}")

    return tp_sl_info

def open_order(api_key, api_secret, symbol, action, usdt_amount, leverage, tp_type="none", tp_value=0, sl_type="none", sl_value=0):
    try:
        exchange = get_exchange(api_key, api_secret)
        exchange.set_leverage(leverage, symbol)
        account = get_account_info(exchange)
        current_pos = get_current_position(exchange, symbol)
        msgs = []

        pre_info = (
            f"\n--- 开单前状态 ---\n"
            f"总余额: {account['total']} USDT\n"
            f"可用余额: {account['free']} USDT\n"
            f"已占用: {account['used']} USDT"
        )

        if current_pos:
            current_side = current_pos["side"]
            current_amount = current_pos["amount"]
            pre_info += (
                f"\n当前持仓: {current_side} {current_amount}\n"
                f"开仓价: {current_pos['entry_price']}\n"
                f"未实现盈亏: {round(current_pos['pnl'], 4)} USDT"
            )
            if action == "buy" and current_side == "long":
                return {"error": "已有多单，跳过"}
            if action == "sell" and current_side == "short":
                return {"error": "已有空单，跳过"}
            if action == "buy" and current_side == "short":
                exchange.create_market_buy_order(symbol, current_amount)
                msgs.append(f"已平空单 {current_amount}")
            if action == "sell" and current_side == "long":
                exchange.create_market_sell_order(symbol, current_amount)
                msgs.append(f"已平多单 {current_amount}")
        else:
            pre_info += "\n当前持仓: 无"

        ticker = exchange.fetch_ticker(symbol)
        price = ticker["last"]
        amount = (usdt_amount * leverage) / price
        amount = exchange.amount_to_precision(symbol, amount)
        amount = float(amount)

        if amount <= 0:
            return {"error": "数量为0，金额太小"}

        if action == "buy":
            order = exchange.create_market_buy_order(symbol, amount)
        elif action == "sell":
            order = exchange.create_market_sell_order(symbol, amount)
        else:
            return {"error": f"未知操作: {action}"}

        tp_price, sl_price = calc_tp_sl_price(action, price, tp_type, tp_value, sl_type, sl_value)
        tp_sl_info = place_tp_sl_orders(exchange, symbol, action, amount, tp_price, sl_price)
        account_after = get_account_info(exchange)
        fee = get_trade_fees(exchange, symbol, limit=3)

        result = {
            "success": True,
            "order_id": order["id"],
            "symbol": symbol,
            "action": action,
            "amount": amount,
            "price": price,
            "fee": fee,
            "pre_info": pre_info,
            "post_total": account_after["total"],
            "post_free": account_after["free"],
            "post_used": account_after["used"],
            "tp_sl_info": tp_sl_info,
        }

        if msgs:
            result["extra"] = " | ".join(msgs)
        return result

    except Exception as e:
        return {"error": str(e)}

def close_position(api_key, api_secret, symbol):
    try:
        exchange = get_exchange(api_key, api_secret)
        current_pos = get_current_position(exchange, symbol)
        if not current_pos:
            return {"error": "没有持仓"}

        account_before = get_account_info(exchange)
        pnl_before = current_pos["pnl"]
        entry_price = current_pos["entry_price"]
        side = current_pos["side"]
        amt = current_pos["amount"]

        if side == "long":
            order = exchange.create_market_sell_order(symbol, amt)
        else:
            order = exchange.create_market_buy_order(symbol, amt)

        try:
            exchange.cancel_all_orders(symbol)
        except:
            pass

        account_after = get_account_info(exchange)
        fee = get_trade_fees(exchange, symbol, limit=3)
        realized_pnl = round(account_after["total"] - account_before["total"], 4)

        return {
            "success": True,
            "closed": side,
            "amount": amt,
            "entry_price": entry_price,
            "pnl": round(pnl_before, 4),
            "realized_pnl": realized_pnl,
            "fee": fee,
            "total_after": account_after["total"],
            "free_after": account_after["free"],
        }
    except Exception as e:
        return {"error": str(e)}

def _run_async(coro):
    def _run():
        loop = asyncio.new_event_loop()
        asyncio.set_event_loop(loop)
        try:
            return loop.run_until_complete(coro)
        finally:
            loop.close()
    future = _executor.submit(_run)
    return future.result(timeout=30)

        logger.info("order_books type: %s", type(result))

        books = None
        if hasattr(result, 'order_books'):
            books = result.order_books
        elif hasattr(result, 'data'):
            books = result.data
        elif isinstance(result, list):
            books = result

        if books:
            for ob in books:
                idx = None
                sym = None
                if hasattr(ob, 'market_index'):
                    idx = int(ob.market_index)
                elif hasattr(ob, 'index'):
                    idx = int(ob.index)
                elif isinstance(ob, dict):
                    idx = int(ob.get('market_index', ob.get('index', -1)))

                if hasattr(ob, 'symbol'):
                    sym = str(ob.symbol)
                elif hasattr(ob, 'ticker'):
                    sym = str(ob.ticker)
                elif isinstance(ob, dict):
                    sym = ob.get('symbol', ob.get('ticker', ''))

                if idx is not None and sym:
                    base = sym.upper().replace("/", "").replace("-", "").replace("_", "")
                    base = base.replace("PERP", "").replace("USD", "").replace("USDT", "").replace("USDC", "")
                    _market_cache[base] = idx
                    logger.info("Market: %s -> %d", base, idx)

        await api_client.close()
    except Exception as e:
        logger.warning("Load markets failed: %s", e)
        defaults = {"ETH": 0, "BTC": 1, "SOL": 2, "ARB": 3, "OP": 4, "AVAX": 5, "DOGE": 6}
        for k, v in defaults.items():
            if k not in _market_cache:
                _market_cache[k] = v

def _ensure_markets():
    if not _market_cache:
        try:
            _run_async(_load_markets_async())
        except Exception as e:
            logger.warning("Market load timeout: %s", e)
            defaults = {"ETH": 0, "BTC": 1, "SOL": 2, "ARB": 3, "OP": 4, "AVAX": 5, "DOGE": 6}
            _market_cache.update(defaults)

def _symbol_to_base(symbol):
    s = symbol.upper().replace("/", "").replace(":USDT", "").replace("-", "")
    s = s.replace(".P", "").replace("PERP", "")
    if s.endswith("USDT"):
        s = s[:-4]
    elif s.endswith("USDC"):
        s = s[:-4]
    elif s.endswith("USD"):
        s = s[:-3]
    return s

def _symbol_to_market_index(symbol):
    _ensure_markets()
    base = _symbol_to_base(symbol)
    idx = _market_cache.get(base)
    if idx is None:
        logger.error("Unknown symbol: %s, cache: %s", base, _market_cache)
    return idx

async def _get_balance_async(account_index):
    api_client = lighter.ApiClient(configuration=lighter.Configuration(host=_get_url()))
    account_api = lighter.AccountApi(api_client)
    try:
        resp = await account_api.account(by="index", value=str(int(account_index)))
        if not resp.accounts or len(resp.accounts) == 0:
            return {"error": "账户不存在"}
        acc = resp.accounts[0]
        total = float(acc.collateral or 0)
        free = float(acc.available_balance or 0)
        used = total - free
        logger.info("Lighter余额: total=%s free=%s", total, free)
        return {"total": round(total, 2), "free": round(free, 2), "used": round(abs(used), 2)}
    except Exception as e:
        logger.error("查询余额失败: %s", e)
        return {"error": str(e)}
    finally:
        await api_client.close()

async def _get_positions_async(account_index):
    api_client = lighter.ApiClient(configuration=lighter.Configuration(host=_get_url()))
    account_api = lighter.AccountApi(api_client)
    try:
        resp = await account_api.account(by="index", value=str(int(account_index)))
        if not resp.accounts or len(resp.accounts) == 0:
            return []
        acc = resp.accounts[0]
        positions = []
        if not acc.positions:
            return positions

        for p in acc.positions:
            pos_size = float(p.position or 0)
            if pos_size == 0:
                continue
            market_idx = int(p.market_id)
            entry_price = float(p.avg_entry_price or 0)
            pnl = float(p.unrealized_pnl or 0)
            margin = float(p.allocated_margin or 0)
            sym_name = (p.symbol + "/USDT:USDT") if p.symbol else "UNKNOWN"
            side = "long" if pos_size > 0 else "short"
            notional = abs(pos_size) * entry_price
            positions.append({
                "symbol": sym_name,
                "side": side,
                "amount": abs(pos_size),
                "entry_price": round(entry_price, 4),
                "pnl": round(pnl, 4),
                "notional": round(notional, 2),
                "margin": round(margin, 2),
                "market_index": market_idx,
            })
        return positions
    except Exception as e:
        logger.error("查询持仓失败: %s", e)
        return {"error": str(e)}
    finally:
        await api_client.close()

async def _open_order_async(private_key, account_index, api_key_index, symbol, action, usdt_amount, leverage, tp_type, tp_value, sl_type, sl_value):
    market_index = _symbol_to_market_index(symbol)
    if market_index is None:
        return {"error": "不支持的币种: " + symbol}

    client = _create_signer(private_key, account_index, api_key_index)

    try:
        tx, tx_hash, err = await client.update_leverage(
            market_index=market_index,
            leverage=int(leverage),
            margin_mode=client.CROSS_MARGIN_MODE,
        )
        if err:
            logger.warning("Set leverage warning: %s", err)

        positions = _run_async(_get_positions_async(account_index))
        if not isinstance(positions, list):
            positions = []

        current_pos = None
        for p in positions:
            if p.get("market_index") == market_index:
                current_pos = p
                break

        msgs = []
        if current_pos:
            current_side = current_pos["side"]
            current_amount = current_pos["amount"]

            if action == "buy" and current_side == "long":
                return {"error": "已有多单，跳过"}
            if action == "sell" and current_side == "short":
                return {"error": "已有空单，跳过"}

            is_ask = True if current_side == "long" else False
            close_quote = current_amount * current_pos["entry_price"]
            tx, tx_hash, err = await client.create_market_order_quote_amount(
                market_index=market_index,
                client_order_index=0,
                quote_amount=close_quote,
                max_slippage=0.005,
                is_ask=is_ask,
            )
            if err:
                logger.warning("Close existing position: %s", err)
            else:
                msgs.append("已平" + ("多" if current_side == "long" else "空") + "单")
            await asyncio.sleep(1)

        quote_amount = usdt_amount * leverage
        is_ask = True if action == "sell" else False
        tx, tx_hash, err = await client.create_market_order_quote_amount(
            market_index=market_index,
            client_order_index=0,
            quote_amount=quote_amount,
            max_slippage=0.005,
            is_ask=is_ask,
        )
        if err:
            return {"error": "下单失败: " + str(err)}

        logger.info("Order tx_hash: %s", tx_hash)

        await asyncio.sleep(2)
        new_positions = _run_async(_get_positions_async(account_index))
        filled_price = 0
        filled_amount = 0
        if isinstance(new_positions, list):
            for p in new_positions:
                if p.get("market_index") == market_index:
                    filled_price = p["entry_price"]
                    filled_amount = p["amount"]
                    break

        tp_sl_info = []
        if filled_price > 0 and filled_amount > 0:
            tp_sl_info = await _set_tp_sl(client, market_index, action, filled_price, tp_type, tp_value, sl_type, sl_value)

        result = {
            "success": True,
            "order_id": str(tx_hash) if tx_hash else "unknown",
            "symbol": symbol,
            "action": action,
            "amount": filled_amount,
            "price": filled_price,
            "fee": 0,
            "tp_sl_info": tp_sl_info,
        }
        if msgs:
            result["extra"] = " | ".join(msgs)
        return result

    except Exception as e:
        logger.error("Open order error: %s", e)
        return {"error": str(e)}

async def _set_tp_sl(client, market_index, action, entry_price, tp_type, tp_value, sl_type, sl_value):
    info = []
    tp_price = None
    sl_price = None

    if tp_type == "percent" and tp_value > 0:
        if action == "buy":
            tp_price = entry_price * (1 + tp_value / 100)
        else:
            tp_price = entry_price * (1 - tp_value / 100)
    elif tp_type == "amount" and tp_value > 0:
        if action == "buy":
            tp_price = entry_price + tp_value
        else:
            tp_price = entry_price - tp_value

    if sl_type == "percent" and sl_value > 0:
        if action == "buy":
            sl_price = entry_price * (1 - sl_value / 100)
        else:
            sl_price = entry_price * (1 + sl_value / 100)
    elif sl_type == "amount" and sl_value > 0:
        if action == "buy":
            sl_price = entry_price - sl_value
        else:
            sl_price = entry_price + sl_value

    if not tp_price and not sl_price:
        return info

    try:
        from lighter.signer_client import CreateOrderTxReq

        is_close_ask = 1 if action == "buy" else 0
        orders = []

        if tp_price:
            tp_price_scaled = int(tp_price * 100)
            trigger_tp = tp_price_scaled
            orders.append(CreateOrderTxReq(
                MarketIndex=market_index,
                ClientOrderIndex=0,
                BaseAmount=0,
                Price=tp_price_scaled,
                IsAsk=is_close_ask,
                Type=client.ORDER_TYPE_TAKE_PROFIT_LIMIT,
                TimeInForce=client.ORDER_TIME_IN_FORCE_GOOD_TILL_TIME,
                ReduceOnly=1,
                TriggerPrice=trigger_tp,
                OrderExpiry=-1,
            ))
            info.append("止盈价: " + str(round(tp_price, 2)))

        if sl_price:
            sl_price_scaled = int(sl_price * 100)
            trigger_sl = sl_price_scaled
            orders.append(CreateOrderTxReq(
                MarketIndex=market_index,
                ClientOrderIndex=0,
                BaseAmount=0,
                Price=sl_price_scaled,
                IsAsk=is_close_ask,
                Type=client.ORDER_TYPE_STOP_LOSS_LIMIT,
                TimeInForce=client.ORDER_TIME_IN_FORCE_GOOD_TILL_TIME,
                ReduceOnly=1,
                TriggerPrice=trigger_sl,
                OrderExpiry=-1,
            ))
            info.append("止损价: " + str(round(sl_price, 2)))

        if len(orders) >= 2:
            tx, tx_hash, err = await client.create_grouped_orders(
                grouping_type=client.GROUPING_TYPE_ONE_CANCELS_THE_OTHER,
                orders=orders,
            )
        elif len(orders) == 1:
            tx, tx_hash, err = await client.create_grouped_orders(
                grouping_type=client.GROUPING_TYPE_ONE_CANCELS_THE_OTHER,
                orders=orders,
            )

        if err:
            info.append("TP/SL设置警告: " + str(err))
    except Exception as e:
        info.append("TP/SL设置失败: " + str(e))
        logger.error("TP/SL error: %s", e)

    return info

async def _close_position_async(private_key, account_index, api_key_index, symbol):
    market_index = _symbol_to_market_index(symbol)
    if market_index is None:
        return {"error": "不支持的币种: " + symbol}

    client = _create_signer(private_key, account_index, api_key_index)

    try:
        positions = _run_async(_get_positions_async(account_index))
        if not isinstance(positions, list):
            return {"error": "查询持仓失败"}

        target = None
        for p in positions:
            if p.get("market_index") == market_index:
                target = p
                break

        if not target:
            return {"error": "没有持仓"}

        side = target["side"]
        amount = target["amount"]
        entry_price = target["entry_price"]
        pnl = target["pnl"]

        is_ask = True if side == "long" else False
        close_quote = amount * entry_price
        tx, tx_hash, err = await client.create_market_order_quote_amount(
            market_index=market_index,
            client_order_index=0,
            quote_amount=close_quote * 1.01,
            max_slippage=0.01,
            is_ask=is_ask,
        )
        if err:
            return {"error": "平仓失败: " + str(err)}

        try:
            ts = int(time.time() * 1000)
            await client.cancel_all_orders(
                time_in_force=client.CANCEL_ALL_TIF_IMMEDIATE,
                timestamp_ms=ts,
            )
        except Exception as e:
            logger.warning("Cancel orders after close: %s", e)

        return {
            "success": True,
            "closed": side,
            "amount": amount,
            "entry_price": entry_price,
            "pnl": round(pnl, 4),
            "price": entry_price,
        }
    except Exception as e:
        logger.error("Close position error: %s", e)
        return {"error": str(e)}

    tg_app.add_handler(CallbackQueryHandler(alert_callback, pattern="^alert"))

    tg_app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))

    flask_thread = Thread(target=run_flask, daemon=True)
    flask_thread.start()

    logger.info("Bot started!")
    tg_app.run_polling(drop_pending_updates=True)

if __name__ == "__main__":
    main()
